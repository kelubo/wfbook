# 区块链基础

## 概述

区块链是一种分布式账本技术，通过去中心化的方式实现数据的安全存储和交易验证。

---

## 一、区块链核心概念

### 1.1 区块链定义

区块链是由多个区块组成的链式数据结构，每个区块包含：
- **区块头**：包含前一个区块的哈希、时间戳、难度值等
- **区块体**：包含交易数据
- **默克尔根**：交易数据的哈希摘要

### 1.2 区块链类型

| 类型 | 特点 | 代表 |
|------|------|------|
| 公有链 | 完全去中心化，任何人可参与 | Bitcoin、Ethereum |
| 私有链 | 中心化管理，权限控制 | Hyperledger Fabric |
| 联盟链 | 多机构共同管理 | R3 Corda |

## 二、密码学基础

### 2.1 哈希函数

```python
import hashlib

def calculate_hash(data):
    """计算SHA-256哈希"""
    if isinstance(data, str):
        data = data.encode('utf-8')
    return hashlib.sha256(data).hexdigest()

# 使用示例
data = "Hello, Blockchain!"
hash_result = calculate_hash(data)
print(f"SHA-256: {hash_result}")

# 验证哈希的单向性
original_data = "Blockchain"
modified_data = "Blockchain1"
print(f"Original hash: {calculate_hash(original_data)}")
print(f"Modified hash: {calculate_hash(modified_data)}")
```

### 2.2 数字签名

```python
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import rsa, padding

# 生成RSA密钥对
private_key = rsa.generate_private_key(
    public_exponent=65537,
    key_size=2048
)
public_key = private_key.public_key()

# 签名数据
message = b"Transaction data"
signature = private_key.sign(
    message,
    padding.PSS(
        mgf=padding.MGF1(hashes.SHA256()),
        salt_length=padding.PSS.MAX_LENGTH
    ),
    hashes.SHA256()
)

# 验证签名
try:
    public_key.verify(
        signature,
        message,
        padding.PSS(
            mgf=padding.MGF1(hashes.SHA256()),
            salt_length=padding.PSS.MAX_LENGTH
        ),
        hashes.SHA256()
    )
    print("Signature verified successfully")
except:
    print("Signature verification failed")
```

### 2.3 默克尔树

```python
import hashlib

def merkle_root(hashes):
    """计算默克尔根"""
    if len(hashes) == 1:
        return hashes[0]
    
    # 如果数量是奇数，复制最后一个哈希
    if len(hashes) % 2 == 1:
        hashes.append(hashes[-1])
    
    # 递归计算
    new_hashes = []
    for i in range(0, len(hashes), 2):
        combined = hashes[i] + hashes[i+1]
        new_hashes.append(hashlib.sha256(combined.encode()).hexdigest())
    
    return merkle_root(new_hashes)

# 使用示例
transactions = ["tx1", "tx2", "tx3", "tx4"]
transaction_hashes = [hashlib.sha256(tx.encode()).hexdigest() for tx in transactions]
root = merkle_root(transaction_hashes)
print(f"Merkle Root: {root}")
```

## 三、共识算法

### 3.1 工作量证明 (PoW)

```python
import hashlib
import time

def proof_of_work(previous_block, transactions, difficulty=4):
    """工作量证明"""
    nonce = 0
    target = '0' * difficulty
    
    while True:
        block_data = f"{previous_block}{transactions}{nonce}"
        hash_result = hashlib.sha256(block_data.encode()).hexdigest()
        
        if hash_result[:difficulty] == target:
            return nonce, hash_result
        
        nonce += 1

# 使用示例
previous_block = "0000a1b2c3d4e5f6"
transactions = ["tx1", "tx2", "tx3"]

start_time = time.time()
nonce, block_hash = proof_of_work(previous_block, str(transactions), difficulty=4)
end_time = time.time()

print(f"Nonce: {nonce}")
print(f"Block Hash: {block_hash}")
print(f"Time taken: {end_time - start_time:.2f} seconds")
```

### 3.2 权益证明 (PoS)

```python
import random
import hashlib

class PoSNode:
    def __init__(self, node_id, stake):
        self.node_id = node_id
        self.stake = stake
    
    def __repr__(self):
        return f"Node {self.node_id} (Stake: {self.stake})"

def proof_of_stake(nodes, previous_block, transactions):
    """权益证明"""
    total_stake = sum(node.stake for node in nodes)
    
    # 根据权益计算权重
    weights = [node.stake / total_stake for node in nodes]
    
    # 随机选择验证者（权重越高，概率越大）
    selected_node = random.choices(nodes, weights=weights)[0]
    
    # 生成区块哈希
    block_data = f"{previous_block}{transactions}{selected_node.node_id}"
    block_hash = hashlib.sha256(block_data.encode()).hexdigest()
    
    return selected_node, block_hash

# 使用示例
nodes = [
    PoSNode("Node1", 100),
    PoSNode("Node2", 200),
    PoSNode("Node3", 300),
    PoSNode("Node4", 400)
]

previous_block = "0000a1b2c3d4e5f6"
transactions = ["tx1", "tx2", "tx3"]

selected_node, block_hash = proof_of_stake(nodes, previous_block, str(transactions))
print(f"Selected validator: {selected_node}")
print(f"Block Hash: {block_hash}")
```

### 3.3 实用拜占庭容错 (PBFT)

```python
class PBFTNode:
    def __init__(self, node_id, is_primary=False):
        self.node_id = node_id
        self.is_primary = is_primary
        self.state = "prepare"  # prepare, commit, reply
        self.messages = []
    
    def send_prepare(self, view, sequence, digest):
        """发送Prepare消息"""
        self.messages.append({
            "type": "prepare",
            "view": view,
            "sequence": sequence,
            "digest": digest,
            "sender": self.node_id
        })
    
    def send_commit(self, view, sequence, digest):
        """发送Commit消息"""
        self.messages.append({
            "type": "commit",
            "view": view,
            "sequence": sequence,
            "digest": digest,
            "sender": self.node_id
        })

def pbft_consensus(nodes, view, sequence, transaction):
    """PBFT共识算法"""
    # 计算交易摘要
    transaction_digest = hashlib.sha256(str(transaction).encode()).hexdigest()
    
    # 主节点广播Prepare消息
    primary_node = [n for n in nodes if n.is_primary][0]
    primary_node.send_prepare(view, sequence, transaction_digest)
    
    # 其他节点验证并广播Commit消息
    commit_count = 0
    for node in nodes:
        if not node.is_primary:
            # 验证Prepare消息
            if len([m for m in node.messages if m['type'] == 'prepare']) >= 2:
                node.send_commit(view, sequence, transaction_digest)
                commit_count += 1
    
    # 检查是否达成共识（需要2f+1个Commit消息）
    if commit_count >= 2 * (len(nodes) // 3) + 1:
        print(f"Consensus reached for transaction {transaction}")
        return True
    else:
        print(f"Consensus failed")
        return False

# 使用示例
nodes = [
    PBFTNode("Node1", is_primary=True),
    PBFTNode("Node2"),
    PBFTNode("Node3"),
    PBFTNode("Node4"),
    PBFTNode("Node5")
]

pbft_consensus(nodes, view=1, sequence=100, transaction="Transfer 100 coins")
```

## 四、区块链数据结构

### 4.1 区块结构

```python
import hashlib
import time

class Block:
    def __init__(self, index, previous_hash, transactions, timestamp=None):
        self.index = index
        self.previous_hash = previous_hash
        self.transactions = transactions
        self.timestamp = timestamp or time.time()
        self.nonce = 0
        self.hash = self.calculate_hash()
    
    def calculate_hash(self):
        """计算区块哈希"""
        block_string = f"{self.index}{self.previous_hash}{self.transactions}{self.timestamp}{self.nonce}"
        return hashlib.sha256(block_string.encode()).hexdigest()
    
    def mine_block(self, difficulty):
        """挖矿"""
        target = '0' * difficulty
        while self.hash[:difficulty] != target:
            self.nonce += 1
            self.hash = self.calculate_hash()
        print(f"Block mined: {self.hash}")
    
    def __repr__(self):
        return f"Block #{self.index} - Hash: {self.hash[:10]}..."
```

### 4.2 区块链结构

```python
class Blockchain:
    def __init__(self):
        self.chain = []
        self.pending_transactions = []
        self.create_genesis_block()
    
    def create_genesis_block(self):
        """创建创世区块"""
        genesis_block = Block(0, "0", ["Genesis Block"])
        self.chain.append(genesis_block)
    
    @property
    def last_block(self):
        return self.chain[-1]
    
    def add_transaction(self, sender, recipient, amount):
        """添加交易"""
        transaction = {
            "sender": sender,
            "recipient": recipient,
            "amount": amount,
            "timestamp": time.time()
        }
        self.pending_transactions.append(transaction)
    
    def mine_pending_transactions(self, miner_address, difficulty=4):
        """挖矿待处理交易"""
        if not self.pending_transactions:
            print("No pending transactions")
            return
        
        # 创建新区块
        new_block = Block(
            len(self.chain),
            self.last_block.hash,
            self.pending_transactions
        )
        
        # 挖矿
        new_block.mine_block(difficulty)
        
        # 添加到链
        self.chain.append(new_block)
        
        # 奖励矿工
        self.add_transaction("Network", miner_address, 1)
        
        # 清空待处理交易
        self.pending_transactions = []
        
        return new_block
    
    def is_chain_valid(self):
        """验证区块链有效性"""
        for i in range(1, len(self.chain)):
            current_block = self.chain[i]
            previous_block = self.chain[i-1]
            
            # 验证区块哈希
            if current_block.hash != current_block.calculate_hash():
                return False
            
            # 验证前一个区块哈希
            if current_block.previous_hash != previous_block.hash:
                return False
        
        return True

# 使用示例
blockchain = Blockchain()

# 添加交易
blockchain.add_transaction("Alice", "Bob", 50)
blockchain.add_transaction("Bob", "Charlie", 25)

# 挖矿
blockchain.mine_pending_transactions("Miner1")

# 验证区块链
print(f"Blockchain valid: {blockchain.is_chain_valid()}")

# 打印区块链
for block in blockchain.chain:
    print(block)
```

pragma solidity ^0.8.0;

contract SensorData {
    struct Data {
        uint256 timestamp;
        uint256 temperature;
        uint256 humidity;
        address device;
    }
    
    Data[] public sensorData;
    
    event DataAdded(uint256 timestamp, uint256 temperature, uint256 humidity);
    
    function addData(uint256 temperature, uint256 humidity) public {
        sensorData.push(Data({
            timestamp: block.timestamp,
            temperature: temperature,
            humidity: humidity,
            device: msg.sender
        }));
        
        emit DataAdded(block.timestamp, temperature, humidity);
    }
    
    function getDataCount() public view returns (uint256) {
        return sensorData.length;
    }
    
    function getData(uint256 index) public view returns (Data memory) {
        require(index < sensorData.length, "Index out of bounds");
        return sensorData[index];
    }
}
