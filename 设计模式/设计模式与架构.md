# 设计模式与架构

## 概述

设计模式是解决特定问题的通用解决方案，架构设计是构建系统的整体蓝图。本章节介绍常用的设计模式和系统架构设计原则。

### 1.2 SOLID原则

| 原则 | 名称 | 描述 |
|------|------|------|
| S | Single Responsibility | 单一职责：一个类只负责一个功能 |
| O | Open/Closed | 开闭原则：对扩展开放，对修改关闭 |
| L | Liskov Substitution | 里氏替换：子类可以替换父类 |
| I | Interface Segregation | 接口隔离：客户端不应依赖不需要的接口 |
| D | Dependency Inversion | 依赖倒置：依赖抽象而非具体实现 |

## 二、创建型模式

### 2.1 单例模式

```python
class Singleton:
    _instance = None
    
    def __new__(cls, *args, **kwargs):
        if not cls._instance:
            cls._instance = super().__new__(cls, *args, **kwargs)
        return cls._instance
    
    def __init__(self):
        self.data = {}

# 使用示例
singleton1 = Singleton()
singleton2 = Singleton()
print(singleton1 is singleton2)  # True
```

### 2.2 工厂模式

```python
from abc import ABC, abstractmethod

class Product(ABC):
    @abstractmethod
    def operation(self):
        pass

class ConcreteProductA(Product):
    def operation(self):
        return "ConcreteProductA operation"

class ConcreteProductB(Product):
    def operation(self):
        return "ConcreteProductB operation"

class Factory:
    def create_product(self, product_type):
        if product_type == "A":
            return ConcreteProductA()
        elif product_type == "B":
            return ConcreteProductB()
        else:
            raise ValueError(f"Unknown product type: {product_type}")

# 使用示例
factory = Factory()
product_a = factory.create_product("A")
product_b = factory.create_product("B")
print(product_a.operation())
print(product_b.operation())
```

### 2.3 抽象工厂模式

```python
from abc import ABC, abstractmethod

# 抽象产品
class AbstractSensor(ABC):
    @abstractmethod
    def read(self):
        pass

class AbstractProcessor(ABC):
    @abstractmethod
    def process(self, data):
        pass

# 具体产品
class TemperatureSensor(AbstractSensor):
    def read(self):
        return 25.5

class HumiditySensor(AbstractSensor):
    def read(self):
        return 60

class SimpleProcessor(AbstractProcessor):
    def process(self, data):
        return f"Processed: {data}"

class AdvancedProcessor(AbstractProcessor):
    def process(self, data):
        return f"Advanced processing: {data * 2}"

# 抽象工厂
class AbstractFactory(ABC):
    @abstractmethod
    def create_sensor(self):
        pass
    
    @abstractmethod
    def create_processor(self):
        pass

# 具体工厂
class TemperatureFactory(AbstractFactory):
    def create_sensor(self):
        return TemperatureSensor()
    
    def create_processor(self):
        return SimpleProcessor()

class HumidityFactory(AbstractFactory):
    def create_sensor(self):
        return HumiditySensor()
    
    def create_processor(self):
        return AdvancedProcessor()

# 使用示例
factory = TemperatureFactory()
sensor = factory.create_sensor()
processor = factory.create_processor()
print(processor.process(sensor.read()))
```

## 三、结构型模式

### 3.1 适配器模式

```python
class Target:
    def request(self):
        return "Target request"

class Adaptee:
    def specific_request(self):
        return "Adaptee specific request"

class Adapter(Target):
    def __init__(self, adaptee):
        self.adaptee = adaptee
    
    def request(self):
        return f"Adapter: {self.adaptee.specific_request()}"

# 使用示例
adaptee = Adaptee()
adapter = Adapter(adaptee)
print(adapter.request())
```

### 3.2 装饰器模式

```python
class Component:
    def operation(self):
        return "Base operation"

class Decorator(Component):
    def __init__(self, component):
        self.component = component
    
    def operation(self):
        return self.component.operation()

class ConcreteDecoratorA(Decorator):
    def operation(self):
        return f"DecoratorA({self.component.operation()})"

class ConcreteDecoratorB(Decorator):
    def operation(self):
        return f"DecoratorB({self.component.operation()})"

# 使用示例
component = Component()
decorated = ConcreteDecoratorA(ConcreteDecoratorB(component))
print(decorated.operation())
```

### 3.3 代理模式

```python
class Subject:
    def request(self):
        pass

class RealSubject(Subject):
    def request(self):
        return "RealSubject handling request"

class Proxy(Subject):
    def __init__(self):
        self.real_subject = None
    
    def request(self):
        if not self.real_subject:
            self.real_subject = RealSubject()
        
        self.log_access()
        return self.real_subject.request()
    
    def log_access(self):
        print("Proxy: Logging access")

# 使用示例
proxy = Proxy()
print(proxy.request())
```

## 四、行为型模式

### 4.1 观察者模式

```python
class Subject:
    def __init__(self):
        self.observers = []
    
    def attach(self, observer):
        self.observers.append(observer)
    
    def detach(self, observer):
        self.observers.remove(observer)
    
    def notify(self, data):
        for observer in self.observers:
            observer.update(data)

class Observer:
    def update(self, data):
        pass

class ConcreteObserverA(Observer):
    def update(self, data):
        print(f"ObserverA received: {data}")

class ConcreteObserverB(Observer):
    def update(self, data):
        print(f"ObserverB received: {data}")

# 使用示例
subject = Subject()
observer_a = ConcreteObserverA()
observer_b = ConcreteObserverB()

subject.attach(observer_a)
subject.attach(observer_b)
subject.notify("Hello, observers!")

subject.detach(observer_a)
subject.notify("Hello again!")
```

### 4.2 策略模式

```python
from abc import ABC, abstractmethod

class Strategy(ABC):
    @abstractmethod
    def execute(self, data):
        pass

class ConcreteStrategyA(Strategy):
    def execute(self, data):
        return f"StrategyA processing: {data}"

class ConcreteStrategyB(Strategy):
    def execute(self, data):
        return f"StrategyB processing: {data.upper()}"

class Context:
    def __init__(self, strategy):
        self.strategy = strategy
    
    def set_strategy(self, strategy):
        self.strategy = strategy
    
    def do_something(self, data):
        return self.strategy.execute(data)

# 使用示例
context = Context(ConcreteStrategyA())
print(context.do_something("test"))

context.set_strategy(ConcreteStrategyB())
print(context.do_something("test"))
```

### 4.3 责任链模式

```python
from abc import ABC, abstractmethod

class Handler(ABC):
    def __init__(self):
        self.next_handler = None
    
    def set_next(self, handler):
        self.next_handler = handler
        return handler
    
    @abstractmethod
    def handle(self, request):
        if self.next_handler:
            return self.next_handler.handle(request)
        return None

class ConcreteHandlerA(Handler):
    def handle(self, request):
        if request == "A":
            return "HandlerA handled request"
        return super().handle(request)

class ConcreteHandlerB(Handler):
    def handle(self, request):
        if request == "B":
            return "HandlerB handled request"
        return super().handle(request)

class ConcreteHandlerC(Handler):
    def handle(self, request):
        if request == "C":
            return "HandlerC handled request"
        return super().handle(request)

# 使用示例
handler_a = ConcreteHandlerA()
handler_b = ConcreteHandlerB()
handler_c = ConcreteHandlerC()

handler_a.set_next(handler_b).set_next(handler_c)

print(handler_a.handle("A"))
print(handler_a.handle("B"))
print(handler_a.handle("C"))
print(handler_a.handle("D"))
```

## 五、系统架构设计

### 5.1 分层架构

```python
# 数据访问层
class SensorRepository:
    def get_sensor_data(self, sensor_id):
        return {"sensor_id": sensor_id, "data": []}

# 业务逻辑层
class SensorService:
    def __init__(self, repository):
        self.repository = repository
    
    def process_sensor_data(self, sensor_id):
        data = self.repository.get_sensor_data(sensor_id)
        return self._transform_data(data)
    
    def _transform_data(self, data):
        return {"processed": data}

# 表示层
class SensorController:
    def __init__(self, service):
        self.service = service
    
    def get_sensor_data(self, sensor_id):
        return self.service.process_sensor_data(sensor_id)

# 使用示例
repository = SensorRepository()
service = SensorService(repository)
controller = SensorController(service)
print(controller.get_sensor_data("sensor1"))
```

### 5.2 微服务架构

```python
# API网关
class APIGateway:
    def __init__(self):
        self.services = {}
    
    def register_service(self, name, service):
        self.services[name] = service
    
    def route(self, service_name, endpoint, data):
        if service_name in self.services:
            return self.services[service_name].handle(endpoint, data)
        return {"error": "Service not found"}

# 传感器服务
class SensorService:
    def handle(self, endpoint, data):
        if endpoint == "/data":
            return {"sensor_data": data}
        return {"error": "Endpoint not found"}

# 用户服务
class UserService:
    def handle(self, endpoint, data):
        if endpoint == "/profile":
            return {"user_profile": data}
        return {"error": "Endpoint not found"}

# 使用示例
gateway = APIGateway()
gateway.register_service("sensor", SensorService())
gateway.register_service("user", UserService())

print(gateway.route("sensor", "/data", {"device_id": "sensor1"}))
print(gateway.route("user", "/profile", {"user_id": "user1"}))
```

### 5.3 事件驱动架构

```python
import asyncio

class EventBus:
    def __init__(self):
        self.subscribers = {}
    
    def subscribe(self, event_type, callback):
        if event_type not in self.subscribers:
            self.subscribers[event_type] = []
        self.subscribers[event_type].append(callback)
    
    async def publish(self, event_type, data):
        if event_type in self.subscribers:
            for callback in self.subscribers[event_type]:
                await callback(data)

# 事件处理器
async def sensor_data_handler(data):
    print(f"Processing sensor data: {data}")

async def alert_handler(data):
    print(f"Sending alert: {data}")

# 使用示例
event_bus = EventBus()
event_bus.subscribe("sensor_data", sensor_data_handler)
event_bus.subscribe("alert", alert_handler)

asyncio.run(event_bus.publish("sensor_data", {"device": "sensor1", "temp": 25}))
asyncio.run(event_bus.publish("alert", {"type": "high_temp", "value": 35}))
```

## 六、高可用架构

### 6.1 负载均衡

```python
import random

class LoadBalancer:
    def __init__(self):
        self.servers = []
        self.weights = []
    
    def add_server(self, server, weight=1):
        self.servers.append(server)
        self.weights.append(weight)
    
    def round_robin(self):
        if not self.servers:
            return None
        if not hasattr(self, '_rr_index'):
            self._rr_index = 0
        server = self.servers[self._rr_index]
        self._rr_index = (self._rr_index + 1) % len(self.servers)
        return server
    
    def weighted_random(self):
        if not self.servers:
            return None
        total_weight = sum(self.weights)
        random_weight = random.uniform(0, total_weight)
        current_weight = 0
        for server, weight in zip(self.servers, self.weights):
            current_weight += weight
            if random_weight <= current_weight:
                return server
        return self.servers[-1]

# 使用示例
lb = LoadBalancer()
lb.add_server("server1", weight=3)
lb.add_server("server2", weight=2)
lb.add_server("server3", weight=1)

print("Round Robin:")
for _ in range(6):
    print(lb.round_robin())

print("\nWeighted Random:")
for _ in range(6):
    print(lb.weighted_random())
```

### 6.2 故障转移

```python
import time

class FailoverManager:
    def __init__(self):
        self.primary = None
        self.secondary = None
        self.max_retries = 3
    
    def set_servers(self, primary, secondary):
        self.primary = primary
        self.secondary = secondary
    
    def get_server(self):
        for attempt in range(self.max_retries):
            if self._is_healthy(self.primary):
                return self.primary
            time.sleep(1)
        
        print("Primary server failed, switching to secondary")
        return self.secondary
    
    def _is_healthy(self, server):
        try:
            # 健康检查逻辑
            return server.health_check()
        except:
            return False

class Server:
    def __init__(self, name):
        self.name = name
        self.is_healthy = True
    
    def health_check(self):
        return self.is_healthy
    
    def __repr__(self):
        return f"Server({self.name})"

# 使用示例
primary = Server("primary")
secondary = Server("secondary")

failover = FailoverManager()
failover.set_servers(primary, secondary)

print(failover.get_server())

# 模拟主服务器故障
primary.is_healthy = False
print(failover.get_server())
```
