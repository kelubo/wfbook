# 嵌入式开发

## 概述

嵌入式开发是指在嵌入式系统上进行软件开发，嵌入式系统通常是资源受限的计算机系统，用于控制特定设备。

## 嵌入式系统特点

- **资源受限**：内存、CPU、存储有限
- **实时性**：需要在规定时间内完成任务
- **低功耗**：通常电池供电
- **可靠性**：长期稳定运行
- **专用性**：针对特定应用设计

## 嵌入式开发流程

```
需求分析 → 硬件选型 → 软件设计 → 编码实现 → 调试测试 → 部署
```

## 硬件基础

### MCU（微控制器）

#### STM32

```
STM32F1xx：入门级，ARM Cortex-M3
STM32F4xx：高性能，ARM Cortex-M4（带FPU）
STM32L4xx：低功耗，ARM Cortex-M4
STM32H7xx：高性能，ARM Cortex-M7
```

#### ESP32

```
ESP32：WiFi+蓝牙，双核Tensilica LX6
ESP32-C3：RISC-V架构，低成本
ESP32-S3：WiFi+蓝牙，更强大的处理器
```

#### Arduino

```
Arduino Uno：ATmega328P，入门级
Arduino Nano：ATmega328P，小型化
Arduino ESP32：ESP32芯片，WiFi+蓝牙
```

### 外设接口

```
UART：通用异步收发器，串口通信
I2C：两线串行总线，设备间通信
SPI：串行外设接口，高速通信
GPIO：通用输入输出，数字信号
ADC：模数转换器，模拟信号转数字
PWM：脉冲宽度调制，控制电机、LED亮度
```

## 嵌入式C编程

### 寄存器操作

```c
// STM32 GPIO寄存器操作示例
#define GPIOA_BASE 0x40020000
#define GPIOA_MODER *(volatile uint32_t*)(GPIOA_BASE + 0x00)
#define GPIOA_ODR   *(volatile uint32_t*)(GPIOA_BASE + 0x14)

void led_init(void) {
    // PA5设为输出模式
    GPIOA_MODER &= ~(0x03 << (5 * 2));
    GPIOA_MODER |= (0x01 << (5 * 2));
}

void led_on(void) {
    GPIOA_ODR |= (1 << 5);
}

void led_off(void) {
    GPIOA_ODR &= ~(1 << 5);
}
```

### 中断处理

```c
// 外部中断示例
void EXTI0_IRQHandler(void) {
    if (EXTI->PR & (1 << 0)) {
        // 处理中断
        led_toggle();
        // 清除中断标志
        EXTI->PR |= (1 << 0);
    }
}

void exti_init(void) {
    // 配置PA0为外部中断输入
    SYSCFG->EXTICR[0] &= ~(0x0F << 0);
    SYSCFG->EXTICR[0] |= (0x00 << 0);
    
    // 配置上升沿触发
    EXTI->RTSR |= (1 << 0);
    
    // 使能中断
    EXTI->IMR |= (1 << 0);
    
    // 配置NVIC
    NVIC_EnableIRQ(EXTI0_IRQn);
    NVIC_SetPriority(EXTI0_IRQn, 0);
}
```

### 定时器

```c
// TIM2定时器示例
void timer_init(void) {
    // 使能TIM2时钟
    RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
    
    // 配置定时器
    TIM2->PSC = 7200 - 1;  // 预分频器，72MHz / 7200 = 10kHz
    TIM2->ARR = 10000 - 1; // 自动重装载值，10kHz / 10000 = 1Hz
    
    // 使能更新中断
    TIM2->DIER |= TIM_DIER_UIE;
    
    // 配置NVIC
    NVIC_EnableIRQ(TIM2_IRQn);
    NVIC_SetPriority(TIM2_IRQn, 1);
    
    // 启动定时器
    TIM2->CR1 |= TIM_CR1_CEN;
}

void TIM2_IRQHandler(void) {
    if (TIM2->SR & TIM_SR_UIF) {
        // 每秒执行一次
        led_toggle();
        // 清除中断标志
        TIM2->SR &= ~TIM_SR_UIF;
    }
}
```

### WiFi连接

```c
#include "esp_wifi.h"
#include "esp_event.h"
#include "freertos/event_groups.h"

#define WIFI_SSID "MY_SSID"
#define WIFI_PASS "MY_PASSWORD"

static EventGroupHandle_t wifi_event_group;
const int CONNECTED_BIT = BIT0;

static void event_handler(void* arg, esp_event_base_t event_base,
                          int32_t event_id, void* event_data) {
    if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_START) {
        esp_wifi_connect();
    } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_DISCONNECTED) {
        esp_wifi_connect();
    } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
        xEventGroupSetBits(wifi_event_group, CONNECTED_BIT);
    }
}

void wifi_init_sta(void) {
    wifi_event_group = xEventGroupCreate();
    
    ESP_ERROR_CHECK(esp_netif_init());
    ESP_ERROR_CHECK(esp_event_loop_create_default());
    esp_netif_create_default_wifi_sta();
    
    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    ESP_ERROR_CHECK(esp_wifi_init(&cfg));
    
    esp_event_handler_instance_t instance_any_id;
    esp_event_handler_instance_t instance_got_ip;
    ESP_ERROR_CHECK(esp_event_handler_instance_register(WIFI_EVENT,
                                                        ESP_EVENT_ANY_ID,
                                                        &event_handler,
                                                        NULL,
                                                        &instance_any_id));
    ESP_ERROR_CHECK(esp_event_handler_instance_register(IP_EVENT,
                                                        IP_EVENT_STA_GOT_IP,
                                                        &event_handler,
                                                        NULL,
                                                        &instance_got_ip));
    
    wifi_config_t wifi_config = {
        .sta = {
            .ssid = WIFI_SSID,
            .password = WIFI_PASS,
        },
    };
    ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
    ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_config));
    ESP_ERROR_CHECK(esp_wifi_start());
    
    xEventGroupWaitBits(wifi_event_group, CONNECTED_BIT, false, true, portMAX_DELAY);
}
```

### 传感器读取

```cpp
#include <DHT.h>

#define DHTPIN 2
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
    Serial.begin(9600);
    dht.begin();
}

void loop() {
    delay(2000);
    
    float h = dht.readHumidity();
    float t = dht.readTemperature();
    
    if (isnan(h) || isnan(t)) {
        Serial.println("Failed to read from DHT sensor!");
        return;
    }
    
    Serial.print("Humidity: ");
    Serial.print(h);
    Serial.print("%\t");
    Serial.print("Temperature: ");
    Serial.print(t);
    Serial.println("°C");
}
```

## 实时操作系统（RTOS）

### FreeRTOS

```c
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"

void task1(void* pvParameters) {
    while (1) {
        printf("Task 1 running\n");
        vTaskDelay(1000 / portTICK_PERIOD_MS);
    }

    while (1) {
        printf("Task 2 running\n");
        vTaskDelay(500 / portTICK_PERIOD_MS);
    }

void app_main(void) {
    xTaskCreate(task1, "task1", 2048, NULL, 5, NULL);
    xTaskCreate(task2, "task2", 2048, NULL, 5, NULL);
    
    vTaskStartScheduler();
}
```

### 任务同步

```c
// 信号量示例
SemaphoreHandle_t xSemaphore;

void task_producer(void* pvParameters) {
    while (1) {
        xSemaphoreGive(xSemaphore);
        vTaskDelay(1000 / portTICK_PERIOD_MS);
    }
}

void task_consumer(void* pvParameters) {
    while (1) {
        xSemaphoreTake(xSemaphore, portMAX_DELAY);
        printf("Semaphore received\n");
    }
}

void app_main(void) {
    xSemaphore = xSemaphoreCreateBinary();
    
    xTaskCreate(task_producer, "producer", 2048, NULL, 5, NULL);
    xTaskCreate(task_consumer, "consumer", 2048, NULL, 5, NULL);
    
    vTaskStartScheduler();
}
```

## 嵌入式通信

### UART通信

```c
// STM32 UART示例
void uart_init(void) {
    // 使能USART1时钟
    RCC->APB2ENR |= RCC_APB2ENR_USART1EN;
    
    // 配置GPIO
    // PA9(TX)设为复用推挽输出
    GPIOA->CRH &= ~(0x0F << 4);
    GPIOA->CRH |= (0x0B << 4);
    // PA10(RX)设为浮空输入
    GPIOA->CRH &= ~(0x0F << 8);
    GPIOA->CRH |= (0x04 << 8);
    
    // 配置USART1
    USART1->BRR = 72000000 / 9600;  // 波特率9600
    USART1->CR1 |= USART_CR1_TE | USART_CR1_RE;  // 使能发送和接收
    USART1->CR1 |= USART_CR1_UE;  // 使能USART1
}

void uart_send_char(char c) {
    while (!(USART1->SR & USART_SR_TXE));
    USART1->DR = c;
}

char uart_receive_char(void) {
    while (!(USART1->SR & USART_SR_RXNE));
    return USART1->DR;
}
```

### I2C通信

```c
// STM32 I2C示例
void i2c_init(void) {
    // 使能I2C1时钟
    RCC->APB1ENR |= RCC_APB1ENR_I2C1EN;
    
    // 配置GPIO
    // PB6(SCL)和PB7(SDA)设为复用开漏输出
    GPIOB->CRL &= ~(0xFF << 24);
    GPIOB->CRL |= (0xEE << 24);
    
    // 配置I2C1
    I2C1->CR1 |= I2C_CR1_SWRST;
    I2C1->CR1 &= ~I2C_CR1_SWRST;
    
    I2C1->CR2 = 36;  // 时钟频率36MHz
    I2C1->CCR = 180;  // 标准模式，100kHz
    I2C1->TRISE = 37;  // 最大上升时间
    
    I2C1->CR1 |= I2C_CR1_PE;  // 使能I2C1
}

void i2c_write(uint8_t addr, uint8_t reg, uint8_t data) {
    // 发送起始条件
    I2C1->CR1 |= I2C_CR1_START;
    while (!(I2C1->SR1 & I2C_SR1_SB));
    
    // 发送设备地址
    I2C1->DR = (addr << 1) | 0;
    while (!(I2C1->SR1 & I2C_SR1_ADDR));
    (void)I2C1->SR2;
    
    // 发送寄存器地址
    I2C1->DR = reg;
    while (!(I2C1->SR1 & I2C_SR1_TXE));
    
    // 发送数据
    I2C1->DR = data;
    while (!(I2C1->SR1 & I2C_SR1_BTF));
    
    // 发送停止条件
    I2C1->CR1 |= I2C_CR1_STOP;
}
```
