# 5G网络架构

## 概述

5G（第五代移动通信技术）是新一代移动通信标准，提供更高的速率、更低的延迟、更大的容量，支持万物互联。

---

## 一、5G核心概念

### 1.1 5G关键指标

| 指标 | 4G | 5G |
|------|----|----|
| 峰值速率 | 1 Gbps | 20 Gbps |
| 用户体验速率 | 10 Mbps | 100 Mbps |
| 延迟 | 50 ms | 1 ms |
| 连接密度 | 100K/km² | 1M/km² |
| 移动性 | 350 km/h | 500 km/h |
| 能效 | 100x | 1000x |

### 2.2 核心网网元功能

| 网元 | 全称 | 功能 |
|------|------|------|
| AMF | Access and Mobility Management Function | 接入管理、移动性管理、认证 |
| SMF | Session Management Function | 会话管理、IP地址分配、QoS控制 |
| UPF | User Plane Function | 用户面转发、流量报告、计费 |
| AUSF | Authentication Server Function | 认证服务 |
| NSSF | Network Slice Selection Function | 网络切片选择 |
| PCF | Policy Control Function | 策略控制、QoS策略 |
| CHF | Charging Function | 计费管理 |

### 3.2 Massive MIMO

```python
import numpy as np

# Massive MIMO信道矩阵
num_antennas = 64
num_users = 8

# 生成信道矩阵
H = np.random.randn(num_antennas, num_users) + 1j * np.random.randn(num_antennas, num_users)

# 预编码矩阵计算
U, S, V = np.linalg.svd(H)
precoding_matrix = V[:, :num_users]

# 波束成形
def beamforming(H, precoding_matrix, data):
    return H @ precoding_matrix @ data

# 计算频谱效率
spectral_efficiency = np.log2(np.linalg.det(np.eye(num_users) + np.linalg.inv(H.conj().T @ H)))
print(f"频谱效率: {spectral_efficiency:.2f} bits/s/Hz")
```

## 四、5G应用

### 4.1 工业5G

```python
# 工业5G通信示例
import socket
import time

class Industrial5GClient:
    def __init__(self, server_ip, server_port):
        self.server_ip = server_ip
        self.server_port = server_port
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    
    def connect(self):
        self.socket.connect((self.server_ip, self.server_port))
        print(f"Connected to {self.server_ip}:{self.server_port}")
    
    def send_data(self, data):
        start_time = time.time()
        self.socket.sendall(data.encode())
        response = self.socket.recv(1024)
        latency = (time.time() - start_time) * 1000
        print(f"Latency: {latency:.2f} ms")
        return response
    
    def close(self):
        self.socket.close()

# 使用示例
client = Industrial5GClient("192.168.1.100", 8888)
client.connect()

# 发送传感器数据
sensor_data = "{\"device\":\"plc1\",\"temperature\":25.5,\"pressure\":10.2}"
response = client.send_data(sensor_data)
print(f"Response: {response.decode()}")

client.close()
```

### 4.2 车联网

```python
import json
import time

class V2XCommunication:
    def __init__(self):
        self.vehicles = {}
    
    def register_vehicle(self, vehicle_id, position):
        self.vehicles[vehicle_id] = {
            'position': position,
            'speed': 0,
            'direction': 0,
            'timestamp': time.time()
        }
    
    def send_safety_message(self, vehicle_id, message_type, data):
        message = {
            'vehicle_id': vehicle_id,
            'message_type': message_type,
            'data': data,
            'timestamp': time.time(),
            'ttl': 100  # 100ms有效时间
        }
        return json.dumps(message)
    
    def receive_message(self, message):
        data = json.loads(message)
        print(f"Received {data['message_type']} from {data['vehicle_id']}")
        return data

# 使用示例
v2x = V2XCommunication()
v2x.register_vehicle("car_001", {"x": 100, "y": 200})

# 发送碰撞预警
message = v2x.send_safety_message("car_001", "collision_warning", {
    "target_vehicle": "car_002",
    "distance": 50,
    "time_to_collision": 2.5
})

# 接收消息
v2x.receive_message(message)
```

### 4.3 远程医疗

```python
import cv2
import numpy as np
import socket

class TelemedicineClient:
    def __init__(self, server_ip, server_port):
        self.server_ip = server_ip
        self.server_port = server_port
    
    def stream_video(self):
        cap = cv2.VideoCapture(0)
        
        # 设置视频参数
        cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1920)
        cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 1080)
        cap.set(cv2.CAP_PROP_FPS, 30)
        
        # 创建socket连接
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.connect((self.server_ip, self.server_port))
        
        while True:
            ret, frame = cap.read()
            if not ret:
                break
            
            # 压缩视频帧
            encode_param = [int(cv2.IMWRITE_JPEG_QUALITY), 90]
            result, img_encode = cv2.imencode('.jpg', frame, encode_param)
            data = np.array(img_encode).tobytes()
            
            # 发送帧大小和数据
            sock.sendall(len(data).to_bytes(4, 'big'))
            sock.sendall(data)
            
            # 接收医生指令
            instruction = sock.recv(1024)
            if instruction:
                print(f"Doctor instruction: {instruction.decode()}")
        
        cap.release()
        sock.close()

# 使用示例
client = TelemedicineClient("192.168.1.100", 9999)
client.stream_video()
```

def calculate_coverage(area_size, cell_radius):
    """计算覆盖指定区域所需的基站数量"""
    cells_per_km2 = 1 / (3 ** 0.5 * cell_radius ** 2)
    total_cells = area_size * cells_per_km2
    return int(total_cells)

# 示例：计算100km²区域的5G基站数量
