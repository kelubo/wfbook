\section{关键参数详解}

\begin{itemize}
    \item \textbf{allowPublicKeyRetrieval=true}: 最核心的参数。\\
          它允许 JDBC 驱动从服务器请求公钥，以完成 caching\_sha2\_password 认证。
    
    \item \textbf{useSSL=false}: 除错必关。\\
          MySQL 8.x 的高版驱动会默认尝试 SSL 连接，在无证书的测试环境必须显式禁用，否则会导致连接失败。
    
    \item \textbf{characterEncoding=UTF-8}: 解决乱码。\\
          强烈建议保留或添加此参数，以确保中文等特殊字符能被正确处理。
\end{itemize}

\subsection{典型连接URL示例}

\begin{verbatim}
jdbc:mysql://localhost:3306/database_name?
    allowPublicKeyRetrieval=true&
    useSSL=false&
    characterEncoding=UTF-8&
    serverTimezone=UTC
\end{verbatim}

\subsection{参数说明}

\begin{description}
    \item[allowPublicKeyRetrieval] 启用公钥检索，解决 MySQL 8.0+ 的认证问题
    \item[useSSL] 控制是否使用 SSL 加密连接
    \item[characterEncoding] 指定字符编码，确保中文正确处理
    \item[serverTimezone] 设置服务器时区，避免时间转换问题
\end{description}
