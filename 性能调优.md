#### 剪枝

```python
import torch
import torch.nn.utils.prune as prune

# 加载模型
model = torch.load('model.pth')

# 对所有卷积层和线性层进行剪枝
parameters_to_prune = []
for name, module in model.named_modules():
    if isinstance(module, torch.nn.Conv2d) or isinstance(module, torch.nn.Linear):
        parameters_to_prune.append((module, 'weight'))

# 剪枝50%的权重
prune.global_unstructured(
    parameters_to_prune,
    pruning_method=prune.L1Unstructured,
    amount=0.5,
)

# 移除剪枝参数，恢复原始参数名
for module, param in parameters_to_prune:
    prune.remove(module, param)

# 微调剪枝后的模型
optimizer = torch.optim.Adam(model.parameters(), lr=1e-5)
# 训练循环...
```

#### 知识蒸馏

```python
import torch
import torch.nn as nn

# 定义教师模型（预训练的大模型）
teacher_model = torch.load('large_model.pth')
teacher_model.eval()

# 定义学生模型（小模型）
class StudentModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(3, 16, 3)
        self.fc1 = nn.Linear(16 * 222 * 222, 10)
    
    def forward(self, x):
        x = self.conv1(x)
        x = x.view(x.size(0), -1)
        x = self.fc1(x)
        return x

student_model = StudentModel()

# 定义损失函数
class DistillationLoss(nn.Module):
    def __init__(self, alpha=0.5, temperature=3):
        super().__init__()
        self.alpha = alpha
        self.temperature = temperature
        self.ce_loss = nn.CrossEntropyLoss()
    
    def forward(self, student_output, teacher_output, labels):
        # 硬标签损失
        hard_loss = self.ce_loss(student_output, labels)
        
        # 软标签损失（知识蒸馏）
        soft_loss = nn.KLDivLoss()(
            nn.functional.log_softmax(student_output / self.temperature, dim=1),
            nn.functional.softmax(teacher_output / self.temperature, dim=1)
        ) * (self.temperature ** 2)
        
        # 总损失
        return self.alpha * hard_loss + (1 - self.alpha) * soft_loss

loss_fn = DistillationLoss(alpha=0.5, temperature=3)
optimizer = torch.optim.Adam(student_model.parameters(), lr=1e-3)

# 训练循环
for epoch in range(100):
    for data, labels in dataloader:
        student_output = student_model(data)
        with torch.no_grad():
            teacher_output = teacher_model(data)
        
        loss = loss_fn(student_output, teacher_output, labels)
        
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
```

#### TensorRT优化

```python
import torch
import tensorrt as trt

# 加载ONNX模型
builder = trt.Builder(trt.Logger())
network = builder.create_network(1 << int(trt.NetworkDefinitionCreationFlag.EXPLICIT_BATCH))
parser = trt.OnnxParser(network, trt.Logger())

with open('resnet18.onnx', 'rb') as f:
    parser.parse(f.read())

# 配置优化参数
config = builder.create_builder_config()
config.set_memory_pool_limit(trt.MemoryPoolType.WORKSPACE, 1 << 30)  # 1GB

# 构建引擎
engine = builder.build_engine(network, config)

# 保存引擎
with open('resnet18.engine', 'wb') as f:
    f.write(engine.serialize())

# 使用TensorRT推理
import pycuda.driver as cuda
import pycuda.autoinit

context = engine.create_execution_context()
inputs = []
outputs = []
bindings = []

for binding in engine:
    size = trt.volume(engine.get_binding_shape(binding)) * engine.max_batch_size
    dtype = trt.nptype(engine.get_binding_dtype(binding))
    host_mem = cuda.pagelocked_empty(size, dtype)
    device_mem = cuda.mem_alloc(host_mem.nbytes)
    bindings.append(int(device_mem))
    if engine.binding_is_input(binding):
        inputs.append({'host': host_mem, 'device': device_mem})
    else:
        outputs.append({'host': host_mem, 'device': device_mem})

# 执行推理
input_data = np.random.randn(1, 3, 224, 224).astype(np.float32)
np.copyto(inputs[0]['host'], input_data.ravel())

cuda.memcpy_htod(inputs[0]['device'], inputs[0]['host'])
context.execute_v2(bindings)
cuda.memcpy_dtoh(outputs[0]['host'], outputs[0]['device'])

result = outputs[0]['host'].reshape(1, 1000)
```

### 3.3 数据预处理优化

```python
import numpy as np
from PIL import Image
import cv2

# 批量数据预处理
def preprocess_batch(images, target_size=(224, 224)):
    batch = np.zeros((len(images), 3, target_size[0], target_size[1]), dtype=np.float32)
    
    for i, img in enumerate(images):
        # 使用OpenCV调整大小
        img = cv2.resize(img, target_size, interpolation=cv2.INTER_LINEAR)
        
        # 转换为Tensor格式
        img = np.transpose(img, (2, 0, 1))
        
        # 归一化
        img = img / 255.0
        img = (img - [0.485, 0.456, 0.406]) / [0.229, 0.224, 0.225]
        
        batch[i] = img
    
    return batch

# 使用多线程加速
from concurrent.futures import ThreadPoolExecutor

def preprocess_image(img):
    img = cv2.resize(img, (224, 224), interpolation=cv2.INTER_LINEAR)
    img = np.transpose(img, (2, 0, 1))
    img = img / 255.0
    img = (img - [0.485, 0.456, 0.406]) / [0.229, 0.224, 0.225]
    return img

with ThreadPoolExecutor(max_workers=4) as executor:
    processed_images = list(executor.map(preprocess_image, images))
```
