### 1.1 ESP32传感器驱动框架

```cpp
#include <Arduino.h>

// 传感器基类
class Sensor {
protected:
    String name;
    float value;
    bool isReady;
    
public:
    Sensor(String sensorName) : name(sensorName), value(0.0), isReady(false) {}
    
    virtual bool begin() = 0;
    virtual float read() = 0;
    virtual String getName() { return name; }
    virtual bool getReady() { return isReady; }
    
    virtual void calibrate() {}
};

// DHT22驱动实现
#include <DHT.h>

class DHT22Sensor : public Sensor {
private:
    DHT dht;
    int pin;
    
public:
    DHT22Sensor(int sensorPin) : Sensor("DHT22"), dht(sensorPin, DHT22), pin(sensorPin) {}
    
    bool begin() override {
        dht.begin();
        delay(100);
        float initialValue = dht.readTemperature();
        if (!isnan(initialValue)) {
            isReady = true;
            Serial.printf("DHT22 initialized on pin %d\n", pin);
            return true;
        }
        return false;
    }
    
    float read() override {
        if (!isReady) return NAN;
        value = dht.readTemperature();
        return value;
    }
    
    float readHumidity() {
        if (!isReady) return NAN;
        return dht.readHumidity();
    }
};

// BME280驱动实现
#include <Wire.h>
#include <Adafruit_BME280.h>

class BME280Sensor : public Sensor {
private:
    Adafruit_BME280 bme;
    
public:
    BME280Sensor() : Sensor("BME280") {}
    
    bool begin() override {
        if (bme.begin(0x76)) {
            isReady = true;
            Serial.println("BME280 initialized");
            return true;
        }
        return false;
    }
    
    float read() override {
        if (!isReady) return NAN;
        value = bme.readTemperature();
        return value;
    }
    
    float readPressure() {
        if (!isReady) return NAN;
        return bme.readPressure() / 100.0;
    }
    
    float readHumidity() {
        if (!isReady) return NAN;
        return bme.readHumidity();
    }
};

// 传感器管理器
class SensorManager {
private:
    Sensor* sensors[10];
    int sensorCount;
    
public:
    SensorManager() : sensorCount(0) {}
    
    bool addSensor(Sensor* sensor) {
        if (sensorCount >= 10) return false;
        if (sensor->begin()) {
            sensors[sensorCount++] = sensor;
            return true;
        }
        return false;
    }
    
    void readAll() {
        for (int i = 0; i < sensorCount; i++) {
            float value = sensors[i]->read();
            Serial.printf("%s: %.2f\n", sensors[i]->getName().c_str(), value);
        }
    }
    
    Sensor* getSensor(int index) {
        if (index < sensorCount) return sensors[index];
        return nullptr;
    }
};

// 使用示例
SensorManager manager;
DHT22Sensor dht22(4);
BME280Sensor bme280;

void setup() {
    Serial.begin(115200);
    Wire.begin();
    
    manager.addSensor(&dht22);
    manager.addSensor(&bme280);
}

void loop() {
    manager.readAll();
    delay(2000);
}
```

### 1.2 自定义传感器驱动模板

```cpp
#include <Arduino.h>

template <typename T>
class GenericSensor {
private:
    String name;
    T value;
    T minValue;
    T maxValue;
    bool calibrated;
    
    // 原始数据读取函数指针
    T (*rawRead)();
    
    // 校准参数
    T offset;
    T scale;
    
public:
    GenericSensor(String n, T (*readFunc)(), T min, T max) 
        : name(n), rawRead(readFunc), minValue(min), maxValue(max), 
          calibrated(false), offset(0), scale(1) {}
    
    bool calibrate(T knownValue) {
        T raw = rawRead();
        if (raw == 0) return false;
        
        offset = knownValue - raw;
        scale = (maxValue - minValue) / (rawRead() - raw + 0.0001);
        calibrated = true;
        return true;
    }
    
    T read() {
        T raw = rawRead();
        if (!calibrated) return raw;
        return (raw + offset) * scale;
    }
    
    bool isValid(T val) {
        return val >= minValue && val <= maxValue;
    }
    
    String getName() { return name; }
};

}
```

### 2.3 MQTT客户端高级功能

```python
import paho.mqtt.client as mqtt
import json
import time

class MQTTClient:
    def __init__(self, client_id, broker, port=1883, username=None, password=None):
        self.client = mqtt.Client(client_id=client_id, clean_session=False)
        self.client.on_connect = self.on_connect
        self.client.on_message = self.on_message
        self.client.on_disconnect = self.on_disconnect
        
        if username and password:
            self.client.username_pw_set(username, password)
        
        self.broker = broker
        self.port = port
        self.callbacks = {}
    
    def on_connect(self, client, userdata, flags, rc):
        print(f"Connected with result code {rc}")
        # 恢复订阅
        for topic in self.callbacks.keys():
            client.subscribe(topic, qos=1)
    
    def on_message(self, client, userdata, msg):
        topic = msg.topic
        if topic in self.callbacks:
            try:
                payload = json.loads(msg.payload.decode())
                self.callbacks[topic](topic, payload)
            except json.JSONDecodeError:
                self.callbacks[topic](topic, msg.payload.decode())
    
    def on_disconnect(self, client, userdata, rc):
        print(f"Disconnected with result code {rc}")
        if rc != 0:
            print("Unexpected disconnection, reconnecting...")
            self.connect()
    
    def connect(self):
        self.client.connect(self.broker, self.port, 60)
    
    def loop(self):
        self.client.loop_forever()
    
    def publish(self, topic, payload, qos=1, retain=False):
        if isinstance(payload, dict):
            payload = json.dumps(payload)
        self.client.publish(topic, payload, qos=qos, retain=retain)
    
    def subscribe(self, topic, callback, qos=1):
        self.callbacks[topic] = callback
        self.client.subscribe(topic, qos=qos)

# 使用示例
def on_sensor_data(topic, data):
    print(f"Received from {topic}: {data}")

client = MQTTClient("python_client", "localhost", username="user", password="pass")
client.connect()
client.subscribe("sensor/data", on_sensor_data)
client.publish("sensor/command", {"action": "start"}, qos=1)
client.loop()
```

## 三、时序数据库查询与可视化

### 3.1 InfluxDB高级查询

```python
from influxdb_client import InfluxDBClient, Point
from influxdb_client.client.write_api import SYNCHRONOUS
import pandas as pd

class InfluxDBManager:
    def __init__(self, url, token, org):
        self.client = InfluxDBClient(url=url, token=token, org=org)
        self.write_api = self.client.write_api(write_options=SYNCHRONOUS)
        self.query_api = self.client.query_api()
    
    def write_data(self, bucket, measurement, tags, fields):
        point = Point(measurement)
        for key, value in tags.items():
            point.tag(key, value)
        for key, value in fields.items():
            point.field(key, value)
        self.write_api.write(bucket=bucket, record=point)
    
    def query_last(self, bucket, measurement, field, tags=None):
        query = f'''
            from(bucket: "{bucket}")
                |> range(start: -1h)
                |> filter(fn: (r) => r._measurement == "{measurement}")
                |> filter(fn: (r) => r._field == "{field}")
        '''
        
        if tags:
            for key, value in tags.items():
                query += f'|> filter(fn: (r) => r.{key} == "{value}")'
        
        query += '|> last()'
        
        result = self.query_api.query(query)
        for table in result:
            for record in table.records:
                return record.get_value()
        return None
    
    def query_range(self, bucket, measurement, field, start="-24h", end="now()", tags=None):
        query = f'''
            from(bucket: "{bucket}")
                |> range(start: {start}, stop: {end})
                |> filter(fn: (r) => r._measurement == "{measurement}")
                |> filter(fn: (r) => r._field == "{field}")
        '''
        
        if tags:
            for key, value in tags.items():
                query += f'|> filter(fn: (r) => r.{key} == "{value}")'
        
        result = self.query_api.query(query)
        data = []
        for table in result:
            for record in table.records:
                data.append({
                    "time": record.get_time(),
                    "value": record.get_value()
                })
        return pd.DataFrame(data)
    
    def query_aggregated(self, bucket, measurement, field, interval="5m", function="mean", start="-24h"):
        query = f'''
            from(bucket: "{bucket}")
                |> range(start: {start})
                |> filter(fn: (r) => r._measurement == "{measurement}")
                |> filter(fn: (r) => r._field == "{field}")
                |> aggregateWindow(every: {interval}, fn: {function})
                |> yield(name: "{function}")
        '''
        
        result = self.query_api.query(query)
        data = []
        for table in result:
            for record in table.records:
                data.append({
                    "time": record.get_time(),
                    "value": record.get_value()
                })
        return pd.DataFrame(data)

# 使用示例
db = InfluxDBManager("http://localhost:8086", "your-token", "myorg")

# 写入数据
db.write_data("sensor_data", "dht22", {"device": "esp32"}, {"temperature": 25.5, "humidity": 60})

# 查询最近值
temp = db.query_last("sensor_data", "dht22", "temperature", {"device": "esp32"})
print(f"Current temperature: {temp}")

# 查询24小时数据
df = db.query_range("sensor_data", "dht22", "temperature", start="-24h")
print(df.head())

# 查询聚合数据
df_agg = db.query_aggregated("sensor_data", "dht22", "temperature", interval="1h", function="mean")
print(df_agg.head())
```

```python
import requests

class GrafanaAPI:
    def __init__(self, url, api_key):
        self.url = url
        self.headers = {"Authorization": f"Bearer {api_key}"}
    
    def create_dashboard(self, dashboard_json):
        response = requests.post(
            f"{self.url}/api/dashboards/db",
            headers=self.headers,
            json=dashboard_json
        )
        return response.json()
    
    def get_dashboard(self, uid):
        response = requests.get(
            f"{self.url}/api/dashboards/uid/{uid}",
            headers=self.headers
        )
        return response.json()
    
    def create_datasource(self, datasource_json):
        response = requests.post(
            f"{self.url}/api/datasources",
            headers=self.headers,
            json=datasource_json
        )
        return response.json()
    
    def create_alert(self, alert_json):
        response = requests.post(
            f"{self.url}/api/alerting/grafana/config",
            headers=self.headers,
            json=alert_json
        )
        return response.json()
```

```python
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
from datetime import datetime

def plot_temperature(df):
    fig, ax = plt.subplots(figsize=(12, 6))
    
    # 转换时间格式
    df['time'] = pd.to_datetime(df['time'])
    
    # 绘制折线图
    ax.plot(df['time'], df['value'], marker='o', linestyle='-', color='b')
    
    # 设置X轴格式
    ax.xaxis.set_major_formatter(mdates.DateFormatter('%Y-%m-%d %H:%M'))
    plt.xticks(rotation=45)
    
    # 添加标签和标题
    ax.set_xlabel('时间')
    ax.set_ylabel('温度 (°C)')
    ax.set_title('温度趋势')
    
    # 添加网格
    ax.grid(True)
    
    # 显示图表
    plt.tight_layout()
    plt.show()
```

### 4.1 传感器数据采集服务

```python
import time
import json
from sensor import DHT22Sensor
from mqtt_client import MQTTClient
from influxdb_manager import InfluxDBManager

# 初始化传感器
sensor = DHT22Sensor(4)
sensor.begin()

# 初始化MQTT客户端
mqtt_client = MQTTClient("sensor_service", "localhost", username="sensor", password="pass")
mqtt_client.connect()

# 初始化InfluxDB客户端
db = InfluxDBManager("http://localhost:8086", "your-token", "myorg")

# 数据采集循环
while True:
    try:
        temperature = sensor.read()
        humidity = sensor.readHumidity()
        
        if not sensor.isValid(temperature) or not sensor.isValid(humidity):
            print("Invalid sensor data")
            time.sleep(5)
            continue
        
        # 构建数据
        data = {
            "device": "esp32_dht22",
            "temperature": temperature,
            "humidity": humidity,
            "timestamp": datetime.now().isoformat()
        }
        
        # 发布到MQTT
        mqtt_client.publish("sensor/data", data, qos=1)
        
        # 写入InfluxDB
        db.write_data("sensor_data", "dht22", {"device": "esp32"}, data)
        
        print(f"Published: {data}")
        
    except Exception as e:
        print(f"Error: {e}")
    
    time.sleep(5)
```
