### 步骤一：ESP32传感器节点开发

#### 硬件连接

```
DHT22 VCC → ESP32 3.3V
DHT22 GND → ESP32 GND
DHT22 DATA → ESP32 GPIO4
```

#### 代码实现

```cpp
#include <WiFi.h>
#include <PubSubClient.h>
#include <DHT.h>

// WiFi配置
const char* ssid = "YOUR_SSID";
const char* password = "YOUR_PASSWORD";

// MQTT配置
const char* mqtt_server = "192.168.1.100";
const int mqtt_port = 1883;
const char* mqtt_topic = "sensor/dht22";

// DHT22配置
#define DHTPIN 4
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

// WiFi和MQTT客户端
WiFiClient espClient;
PubSubClient client(espClient);

void setup() {
  Serial.begin(115200);
  
  // 初始化DHT22
  dht.begin();
  
  // 连接WiFi
  setup_wifi();
  
  // 连接MQTT
  client.setServer(mqtt_server, mqtt_port);
  client.setCallback(callback);
}

void setup_wifi() {
  delay(10);
  Serial.println();
  Serial.print("Connecting to ");
  Serial.println(ssid);
  
  WiFi.begin(ssid, password);
  
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  
  Serial.println("");
  Serial.println("WiFi connected");
  Serial.println("IP address: ");
  Serial.println(WiFi.localIP());
}

void callback(char* topic, byte* payload, unsigned int length) {
  Serial.print("Message arrived [");
  Serial.print(topic);
  Serial.print("] ");
  for (int i = 0; i < length; i++) {
    Serial.print((char)payload[i]);
  }
  Serial.println();
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");
    if (client.connect("ESP32_DHT22_Client")) {
      Serial.println("connected");
      client.subscribe("sensor/command");
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      delay(5000);
    }
  }
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop();
  
  // 每5秒采集一次数据
  delay(5000);
  
  // 读取传感器数据
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  
  if (isnan(h) || isnan(t)) {
    Serial.println("Failed to read from DHT sensor!");
    return;
  }
  
  // 构建JSON消息
  String json = "{\"device\":\"esp32_dht22\",\"temperature\":" + String(t) + ",\"humidity\":" + String(h) + ",\"timestamp\":\"" + String(millis()) + "\"}";
  
  // 发布消息
  client.publish(mqtt_topic, json.c_str());
  
  // 打印调试信息
  Serial.print("Temperature: ");
  Serial.print(t);
  Serial.print(" °C\tHumidity: ");
  Serial.print(h);
  Serial.println(" %");
}
```

#### Python数据写入脚本

```python
import paho.mqtt.client as mqtt
from influxdb_client import InfluxDBClient, Point

# MQTT配置
MQTT_BROKER = "localhost"
MQTT_PORT = 1883
MQTT_TOPIC = "sensor/dht22"

# InfluxDB配置
INFLUXDB_URL = "http://localhost:8086"
INFLUXDB_TOKEN = "your-token"
INFLUXDB_ORG = "myorg"
INFLUXDB_BUCKET = "sensor_data"

# 创建InfluxDB客户端
influx_client = InfluxDBClient(url=INFLUXDB_URL, token=INFLUXDB_TOKEN, org=INFLUXDB_ORG)
write_api = influx_client.write_api()

def on_connect(client, userdata, flags, rc):
    print(f"Connected with result code {rc}")
    client.subscribe(MQTT_TOPIC)

def on_message(client, userdata, msg):
    import json
    try:
        data = json.loads(msg.payload.decode())
        
        # 创建InfluxDB点
        point = Point("dht22") \
            .tag("device", data["device"]) \
            .field("temperature", data["temperature"]) \
            .field("humidity", data["humidity"])
        
        # 写入数据
        write_api.write(bucket=INFLUXDB_BUCKET, record=point)
        print(f"Wrote data: {data}")
    except Exception as e:
        print(f"Error: {e}")

# 创建MQTT客户端
mqtt_client = mqtt.Client()
mqtt_client.on_connect = on_connect
mqtt_client.on_message = on_message

mqtt_client.connect(MQTT_BROKER, MQTT_PORT, 60)
mqtt_client.loop_forever()
```

### Python数据采集

#### 安装依赖

```bash
pip install pymodbus
```

#### Modbus TCP客户端

```python
from pymodbus.client import ModbusTcpClient

# 连接PLC
client = ModbusTcpClient('192.168.1.10', port=502)
client.connect()

# 读取保持寄存器（温度）
response = client.read_holding_registers(address=0, count=1, unit=1)
temperature = response.registers[0] / 10.0  # 假设10倍缩放

# 读取保持寄存器（压力）
response = client.read_holding_registers(address=1, count=1, unit=1)
pressure = response.registers[0] / 100.0

# 读取保持寄存器（流量）
response = client.read_holding_registers(address=2, count=1, unit=1)
flow = response.registers[0] / 10.0

print(f"Temperature: {temperature} °C")
print(f"Pressure: {pressure} bar")
print(f"Flow: {flow} m3/h")

client.close()
```

## 项目三：AI传感器数据分析项目

### 项目概述

本项目实现基于AI的传感器数据分析系统，包括：
- 传感器数据收集
- 数据预处理
- AI模型训练
- 异常检测

### 数据收集

```python
import pandas as pd

# 加载传感器数据
data = pd.read_csv("sensor_data.csv")
print(data.head())
```

### 数据预处理

```python
# 处理缺失值
data = data.dropna()

# 处理异常值
Q1 = data['temperature'].quantile(0.25)
Q3 = data['temperature'].quantile(0.75)
IQR = Q3 - Q1
data = data[(data['temperature'] >= Q1 - 1.5*IQR) & (data['temperature'] <= Q3 + 1.5*IQR)]

# 特征工程
data['hour'] = pd.to_datetime(data['timestamp']).dt.hour
data['day_of_week'] = pd.to_datetime(data['timestamp']).dt.dayofweek
```

### 模型训练

```python
from sklearn.model_selection import train_test_split
from sklearn.ensemble import IsolationForest

# 准备数据
X = data[['temperature', 'humidity', 'hour', 'day_of_week']]

# 训练异常检测模型
model = IsolationForest(contamination=0.05)
model.fit(X)

# 预测异常
data['anomaly'] = model.predict(X)
```

### 边缘部署

```python
import joblib

# 保存模型
joblib.dump(model, 'anomaly_model.pkl')

# 加载模型（边缘设备）
model = joblib.load('anomaly_model.pkl')

# 实时检测
def detect_anomaly(sensor_data):
    prediction = model.predict([sensor_data])
    if prediction == -1:
        return True  # 异常
    return False  # 正常
```
