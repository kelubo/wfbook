# 工业物联网

## 概述

工业物联网（IIoT）是物联网技术在工业领域的应用，通过连接工业设备和传感器，实现工业生产的智能化和自动化。

### Modbus协议

```python
# Modbus客户端示例
from pymodbus.client import ModbusTcpClient

client = ModbusTcpClient('192.168.1.10', port=502)
client.connect()

# 读取保持寄存器
response = client.read_holding_registers(address=0, count=10, unit=1)
print(response.registers)

# 写入保持寄存器
client.write_register(address=0, value=100, unit=1)

client.close()
```

### OPC UA协议

```python
# OPC UA客户端示例
from opcua import Client

client = Client("opc.tcp://192.168.1.10:4840")
client.connect()

# 获取节点
node = client.get_node("ns=2;i=1")
value = node.get_value()
print(value)

# 设置节点值
node.set_value(100)

client.disconnect()
```

### 预测维护

```
基于传感器数据预测设备故障
```

```python
# 预测维护示例
import pandas as pd
from sklearn.ensemble import RandomForestClassifier

# 加载历史数据
data = pd.read_csv("machine_data.csv")
X = data[["vibration", "temperature", "pressure"]]
y = data["failure"]

# 训练模型
model = RandomForestClassifier()
model.fit(X, y)

# 预测故障
prediction = model.predict([[5.2, 75, 100]])
print(f"Failure prediction: {prediction[0]}")
```

class EnergyManager:
    def __init__(self):
        self.meters = {}
    
    def read_meter(self, meter_id):
        if meter_id in self.meters:
            return self.meters[meter_id]
        return None
    
    def calculate_cost(self, consumption, rate=0.15):
        return consumption * rate

### 生产优化

```
优化生产流程和效率
```

```python
# 生产优化示例
class ProductionOptimizer:
    def __init__(self):
        self.machines = {}
    
    def add_machine(self, machine_id, efficiency):
        self.machines[machine_id] = {"efficiency": efficiency, "status": "running"}
    
    def optimize(self):
        # 关闭低效率设备
        for machine, info in self.machines.items():
            if info["efficiency"] < 0.6:
                info["status"] = "idle"
```

## 工业物联网安全

### 设备认证

```python
# 设备认证示例
import jwt

def generate_token(device_id):
    payload = {
        'device_id': device_id,
        'exp': datetime.utcnow() + timedelta(hours=24)
    }
    token = jwt.encode(payload, 'secret_key', algorithm='HS256')
    return token

def verify_token(token):
    try:
        payload = jwt.decode(token, 'secret_key', algorithms=['HS256'])
        return payload['device_id']
    except jwt.ExpiredSignatureError:
        return None
```
