## 五、知识图谱构建

### 5.1 知识表示

```python
class KnowledgeGraph:
    def __init__(self):
        self.nodes = {}
        self.edges = []
    
    def add_node(self, node_id, node_type, attributes):
        self.nodes[node_id] = {
            'type': node_type,
            'attributes': attributes
        }
    
    def add_edge(self, from_node, to_node, relation):
        self.edges.append({
            'from': from_node,
            'to': to_node,
            'relation': relation
        })
    
    def query_relations(self, node_id):
        results = []
        for edge in self.edges:
            if edge['from'] == node_id or edge['to'] == node_id:
                results.append(edge)
        return results
    
    def get_node(self, node_id):
        return self.nodes.get(node_id)

# 使用示例
kg = KnowledgeGraph()

# 添加节点
kg.add_node('sensor1', '传感器', {'type': '温湿度', 'model': 'DHT22'})
kg.add_node('mqtt', '协议', {'type': '物联网协议', 'port': 1883})
kg.add_node('influxdb', '数据库', {'type': '时序数据库', 'version': '2.0'})

# 添加关系
kg.add_edge('sensor1', 'mqtt', '使用')
kg.add_edge('mqtt', 'influxdb', '连接')

# 查询关系
print(kg.query_relations('sensor1'))
print(kg.get_node('mqtt'))
```

### 5.2 知识推理

```python
class Reasoner:
    def __init__(self, knowledge_graph):
        self.kg = knowledge_graph
    
    def infer_connections(self, node_id):
        """推断节点之间的间接连接"""
        direct_relations = self.kg.query_relations(node_id)
        indirect_relations = []
        
        for relation in direct_relations:
            other_node = relation['to'] if relation['from'] == node_id else relation['from']
            other_relations = self.kg.query_relations(other_node)
            
            for or_rel in other_relations:
                third_node = or_rel['to'] if or_rel['from'] == other_node else or_rel['from']
                if third_node != node_id:
                    indirect_relations.append({
                        'from': node_id,
                        'through': other_node,
                        'to': third_node,
                        'relations': [relation['relation'], or_rel['relation']]
                    })
        
        return indirect_relations

# 使用示例
reasoner = Reasoner(kg)
inferred = reasoner.infer_connections('sensor1')
print("推断的间接连接:")
for inf in inferred:
    print(f"{inf['from']} -> {inf['through']} -> {inf['to']}")
```

### 5.3 知识搜索

```python
class KnowledgeSearch:
    def __init__(self, knowledge_graph):
        self.kg = knowledge_graph
    
    def search_by_type(self, node_type):
        """按类型搜索节点"""
        results = []
        for node_id, node_data in self.kg.nodes.items():
            if node_data['type'] == node_type:
                results.append({node_id: node_data})
        return results
    
    def search_by_attribute(self, attr_key, attr_value):
        """按属性搜索节点"""
        results = []
        for node_id, node_data in self.kg.nodes.items():
            if attr_key in node_data['attributes'] and node_data['attributes'][attr_key] == attr_value:
                results.append({node_id: node_data})
        return results

# 使用示例
searcher = KnowledgeSearch(kg)
print("传感器类型节点:")
print(searcher.search_by_type('传感器'))

print("\n协议类型节点:")
print(searcher.search_by_type('协议'))
```
